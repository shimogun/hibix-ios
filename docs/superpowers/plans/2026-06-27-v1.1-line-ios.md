# Hibix iOS v1.1 LINE通知 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** iOS が contacts/settings をサーバー同期し、email/LINE 同列の連絡先と LINE 連携コード方式（友だち追加→6桁コード→push）を扱えるようにする（タスク#10）。

**Architecture:** 既存 `CheckinService`（APIClient＋Repository を束ねるサービス層）パターンを踏襲。ローカル `Int64 id` は維持し、サーバーUUIDを `server_id` 列で別管理。新規 `ContactsSyncService`（contacts/settings push＋server_id 書き戻し）と `LineLinkService`（issue-code/status）を追加し、編集UI/モード切替/LINE連携画面から駆動する。

**Tech Stack:** Swift / SwiftUI / GRDB.swift / App Attest（DCAppAttestService）/ Swift Testing（`@Suite`/`@Test`/`#expect`）。

## Global Constraints

- 命名: 変数/関数 camelCase、型/コンポーネント PascalCase。未使用 import 削除。
- 非同期の外部API/ストレージアクセスは `do/catch`（無言クラッシュ防止）。
- 禁止: 第三者解析/広告SDK追加、`mood_level`/`memo`/ピクセル履歴のサーバー送信、`com.shimogun.hibix.pro.lifetime` 以外の商品ID、月額サブスク、`app.json`/`eas.json` 相当（`*.xcodeproj`/`*.entitlements`/`Info.plist` の課金・証明書設定）の無断変更、`git push --force`/`rm -rf`、薬機法NG表現（「治療」「改善」「予防」）をコード/コメント/通知本文に入れない。

- LINE の `expires_at` / `code_expires_at` は **UNIX epoch 秒（Int）**。他APIのISO8601 Dateと異なるため Int でデコードして変換する。

import Foundation

/// LINE 連携状態（backend `line_link_status` のローカルキャッシュ）。
enum LineLinkStatus: String, CaseIterable, Codable, Sendable {
    case unlinked
    case pending
    case linked

    var badgeText: String {
        switch self {
        case .unlinked: return "未連携"
        case .pending:  return "連携待ち"
        case .linked:   return "連携済み"
        }
    }

    /// DB 保存値からの復元。未知/nil は unlinked。
    static func fromStoredValue(_ raw: String?) -> LineLinkStatus {
        guard let raw, let value = LineLinkStatus(rawValue: raw) else { return .unlinked }
        return value
    }
}
```

import Foundation
import os.log

/// contacts / settings をサーバーへ同期し、PUT レスポンスの id をローカルへ書き戻す。
/// 失敗は best-effort（次回 resync）。LINE フローからは throwing 版を使う。
@MainActor
final class ContactsSyncService {
    @ObservationIgnored private let apiClient: APIClient
    @ObservationIgnored private let contactsRepo: EmergencyContactsRepository
    @ObservationIgnored private let settings: SettingsRepository
    @ObservationIgnored private let attest: AppAttestClient
    @ObservationIgnored private weak var deletionPending: DeletionPendingCoordinator?

    private static let logger = Logger(subsystem: "com.shimogun.hibix", category: "ContactsSync")

    init(apiClient: APIClient,
         contactsRepo: EmergencyContactsRepository,
         settings: SettingsRepository,
         attest: AppAttestClient,
         deletionPending: DeletionPendingCoordinator? = nil) {
        self.apiClient = apiClient
        self.contactsRepo = contactsRepo
        self.settings = settings
        self.attest = attest
        self.deletionPending = deletionPending
    }

    func attach(deletionPending: DeletionPendingCoordinator) { self.deletionPending = deletionPending }

    func syncContacts() async {
        do { try await syncContactsThrowing() }
        catch let error as APIError {
            Self.logger.error("syncContacts failed: \(error.localizedDescription, privacy: .public)")
            if error.isDeletionPending { deletionPending?.report(error) }
        } catch {
            Self.logger.error("syncContacts failed: \(error.localizedDescription, privacy: .public)")
        }
    }

    func syncContactsThrowing() async throws {
        guard attest.isSupported, attest.isRegistered else {
            Self.logger.notice("Skip contacts sync: attest unavailable")
            return
        }
        let local = try await contactsRepo.list()   // sort_order 昇順
        let body = ContactsPutBody(contacts: local.map { c in
            ContactInputBody(
                id: c.serverID,
                contact_type: c.contactType.rawValue,
                email: c.contactType == .email ? c.email : nil,
                label: c.label
            )
        })
        let resp: ContactsResponse = try await apiClient.request(.contacts(body))
        // リクエスト順＝レスポンス sort_order 順。index 対応で server_id を書き戻す。
        let pairs: [(localID: Int64, serverID: String)] = zip(local, resp.contacts).map {
            (localID: $0.0.id, serverID: $0.1.id)
        }
        if !pairs.isEmpty { try await contactsRepo.updateServerMapping(pairs) }
    }

    func syncSettings(watchMode: String, watchDays: Int) async throws {
        guard attest.isSupported, attest.isRegistered else { return }
        let body = SettingsPatchBody(watch_days: watchDays, watch_mode: watchMode)
        let _: SettingsResponse = try await apiClient.request(.settings(body))
    }

    func resyncOnLaunch() async { await syncContacts() }
}
```

import Foundation
import os.log

@MainActor
final class LineLinkService {
    struct LineLinkCode { let code: String; let expiresAt: Date; let addFriendURL: URL? }
    enum LineLinkError: Error { case notSynced }

    @ObservationIgnored private let apiClient: APIClient
    @ObservationIgnored private let contactsRepo: EmergencyContactsRepository
    @ObservationIgnored private let contactsSync: ContactsSyncService
    private static let logger = Logger(subsystem: "com.shimogun.hibix", category: "LineLink")

    init(apiClient: APIClient, contactsRepo: EmergencyContactsRepository, contactsSync: ContactsSyncService) {
        self.apiClient = apiClient
        self.contactsRepo = contactsRepo
        self.contactsSync = contactsSync
    }

    private func resolveServerID(localContactID: Int64) async throws -> String {
        if let sid = try await contactsRepo.list().first(where: { $0.id == localContactID })?.serverID { return sid }
        try await contactsSync.syncContactsThrowing()
        guard let sid = try await contactsRepo.list().first(where: { $0.id == localContactID })?.serverID else {
            throw LineLinkError.notSynced
        }
        return sid
    }

    func issueCode(localContactID: Int64) async throws -> LineLinkCode {
        let serverID = try await resolveServerID(localContactID: localContactID)
        let resp: LineIssueCodeResponse = try await apiClient.request(.lineIssueCode(serverContactID: serverID))
        try await contactsRepo.updateLineLinkStatus(localID: localContactID, status: .pending)
        return LineLinkCode(
            code: resp.code,
            expiresAt: Date(timeIntervalSince1970: TimeInterval(resp.expires_at)),
            addFriendURL: resp.add_friend_url.flatMap(URL.init(string:))
        )
    }

    func fetchStatus(localContactID: Int64) async throws -> LineLinkStatus {
        let serverID = try await resolveServerID(localContactID: localContactID)
        let resp: LineStatusResponse = try await apiClient.request(.lineStatus(serverContactID: serverID))
        let status = LineLinkStatus.fromStoredValue(resp.status)
        try await contactsRepo.updateLineLinkStatus(localID: localContactID, status: status)
        return status
    }
}

import Foundation
import Observation
import os.log

@MainActor
@Observable
final class LineLinkViewModel {
    enum Phase {
        case loading
        case code(LineLinkService.LineLinkCode)
        case linked
        case expired
        case error(String)
    }

    private(set) var phase: Phase = .loading

    @ObservationIgnored private let service: LineLinkService
    @ObservationIgnored private let localContactID: Int64
    private static let logger = Logger(subsystem: "com.shimogun.hibix", category: "LineLinkVM")

    init(service: LineLinkService, localContactID: Int64) {
        self.service = service
        self.localContactID = localContactID
    }

    func start() async {
        phase = .loading
        do {
            let code = try await service.issueCode(localContactID: localContactID)
            phase = .code(code)
        } catch {
            Self.logger.error("issueCode failed: \(error.localizedDescription, privacy: .public)")
            phase = .error("連携コードの発行に失敗しました。通信状況を確認して再度お試しください")
        }
    }

    func reissue() async { await start() }

    /// linked になったら true。呼び出し側が一定間隔で回す。
    func pollOnce() async -> Bool {
        guard case .code(let code) = phase else { return false }
        if code.expiresAt <= Date() { phase = .expired; return false }
        do {
            let status = try await service.fetchStatus(localContactID: localContactID)
            if status == .linked { phase = .linked; return true }
            return false
        } catch {
            Self.logger.error("fetchStatus failed: \(error.localizedDescription, privacy: .public)")
            return false
        }
    }

    var shareText: String {
        guard case .code(let code) = phase else { return "" }
        return "Hibix の見守り通知を LINE で受け取るには、公式アカウントを友だち追加して、このコードをトークに送ってね: \(code.code)"
    }
}
